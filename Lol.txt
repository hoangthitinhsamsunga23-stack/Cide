// ==UserScript==
// @name         🔥 Video Hunter Pro - M3U8 to MP4 Converter
// @namespace    http://tampermonkey.net/
// @version      6.0
// @description  Bắt video, tự động chuyển M3U8 sang MP4, tải mọi video từ mọi trang
// @author       HunterPro
// @match        *://*/*
// @require      https://cdn.jsdelivr.net/npm/hls.js@latest
// @grant        none
// @run-at       document-start
// ==/UserScript==

(function() {
    'use strict';

    let videoSources = new Set();
    let isPanelOpen = false;

    // 1. INTERCEPT NETWORK (Bắt link từ lúc trang web vừa tải)
    const originalFetch = window.fetch;
    window.fetch = function(...args) {
        const url = typeof args[0] === 'string' ? args[0] : (args[0]?.url || '');
        checkAndAddUrl(url);
        return originalFetch.apply(this, args);
    };

    const originalXHROpen = XMLHttpRequest.prototype.open;
    XMLHttpRequest.prototype.open = function(method, url, ...rest) {
        checkAndAddUrl(url);
        return originalXHROpen.apply(this, [method, url, ...rest]);
    };

    function checkAndAddUrl(url) {
        if (!url || typeof url !== 'string' || url.startsWith('blob:')) return;
        if (/\.(mp4|webm|ogg|mov|m3u8|mpd|flv)/i.test(url) || /(video|stream|videoplayback)/i.test(url)) {
            videoSources.add(url);
            updateBadge();
        }
    }

    // 2. SCAN DOM VIDEOS
    function scanDOMVideos() {
        document.querySelectorAll('video').forEach(v => {
            if (v.src && !v.src.startsWith('blob:')) videoSources.add(v.src);
            if (v.currentSrc && !v.currentSrc.startsWith('blob:')) videoSources.add(v.currentSrc);
            v.querySelectorAll('source').forEach(s => { if (s.src) videoSources.add(s.src); });
        });
    }

    // 3. M3U8 TO MP4 CONVERTER (Tải các đoạn .ts và ghép lại)
    async function convertM3U8ToMP4(m3u8Url, btnElement) {
        btnElement.innerText = "⏳ Đang đọc M3U8...";
        btnElement.style.background = "#6366f1";

        try {
            const response = await fetch(m3u8Url);
            const text = await response.text();
            const lines = text.split('\n');
            const tsUrls = [];
            let baseUrl = m3u8Url.substring(0, m3u8Url.lastIndexOf('/') + 1);

            for (let line of lines) {
                line = line.trim();
                if (!line.startsWith('#') && line.length > 0) {
                    tsUrls.push(line.startsWith('http') ? line : baseUrl + line);
                }
            }

            if (tsUrls.length === 0) {
                throw new Error("Không tìm thấy file .ts trong playlist!");
            }

            const blobs = [];
            for (let i = 0; i < tsUrls.length; i++) {
                btnElement.innerText = `⏳ Tải đoạn ${i + 1}/${tsUrls.length}`;
                const tsResp = await fetch(tsUrls[i]);
                const blob = await tsResp.blob();
                blobs.push(blob);
            }

            btnElement.innerText = "📦 Đang ghép video...";
            const finalBlob = new Blob(blobs, { type: 'video/mp4' });
            const blobUrl = URL.createObjectURL(finalBlob);

            const a = document.createElement('a');
            a.href = blobUrl;
            a.download = `video_converted_${Date.now()}.mp4`;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(blobUrl);

            btnElement.innerText = "✅ Tải Xong!";
            btnElement.style.background = "#10b981";

        } catch (e) {
            console.error("Lỗi chuyển đổi M3U8:", e);
            btnElement.innerText = "❌ Lỗi CORS";
            btnElement.style.background = "#ef4444";
            alert("⚠️ Trình duyệt chặn tải do CORS.\nHãy bấm 'Open Link' hoặc 'Copy Link' rồi dán vào IDM / 1DM / XDM để tải!");
        }
    }

    // 4. FORCE DOWNLOAD MP4 DIRECT
    async function forceDownload(url, filename, btnElement) {
        btnElement.innerText = "⏳ Đang xử lý...";
        btnElement.style.opacity = "0.7";
        try {
            const response = await fetch(url, { mode: 'cors', credentials: 'omit' });
            if (!response.ok) throw new Error('Network error');
            const blob = await response.blob();
            const blobUrl = URL.createObjectURL(blob);
            const a = document.createElement('a');
            a.href = blobUrl;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            URL.revokeObjectURL(blobUrl);
            btnElement.innerText = "✅ Xong!";
            btnElement.style.background = "#059669";
        } catch (e) {
            btnElement.innerText = "⚠️ Bấm Open Link";
            btnElement.style.background = "#ef4444";
            alert("Trình duyệt chặn tải trực tiếp. Hãy bấm 'Open Link' để mở video, sau đó Chuột phải -> Save Video As!");
        }
    }

    function copyToClipboard(text) {
        navigator.clipboard.writeText(text).then(() => {
            alert("✅ Đã sao chép link video!\nDán vào IDM, 1DM hoặc MX Player để tải/xem.");
        }).catch(() => {
            const textarea = document.createElement('textarea');
            textarea.value = text;
            document.body.appendChild(textarea);
            textarea.select();
            document.execCommand('copy');
            document.body.removeChild(textarea);
            alert("✅ Đã sao chép link video!");
        });
    }

    function getFileName(url) {
        try {
            let path = new URL(url).pathname;
            let name = path.split('/').pop();
            return name.includes('.') ? name : 'video.mp4';
        } catch(e) { return 'video.mp4'; }
    }

    // 5. GIAO DIỆN (MOBILE UI)
    function createUI() {
        if (document.getElementById('vh-container')) return;

        const css = document.createElement('style');
        css.textContent = `
            #vh-container { position: fixed; bottom: 20px; right: 16px; z-index: 2147483647; font-family: system-ui, sans-serif; }
            #vh-fab { width: 56px; height: 56px; border-radius: 50%; background: linear-gradient(135deg, #f43f5e, #ec4899); border: none; color: #fff; font-size: 26px; box-shadow: 0 4px 15px rgba(244,63,94,0.5); cursor: pointer; position: relative; }
            #vh-badge { position: absolute; top: -5px; right: -5px; background: #10b981; color: #fff; border-radius: 50%; min-width: 20px; height: 20px; font-size: 11px; font-weight: bold; display: flex; align-items: center; justify-content: center; border: 2px solid #fff; display: none; }
            #vh-panel { position: fixed; bottom: 0; left: 0; right: 0; background: #0f172a; color: #e2e8f0; max-height: 85vh; border-radius: 20px 20px 0 0; transform: translateY(100%); transition: transform 0.3s ease; display: flex; flex-direction: column; box-shadow: 0 -10px 40px rgba(0,0,0,0.5); }
            #vh-panel.open { transform: translateY(0); }
            .vh-header { padding: 16px 20px; background: linear-gradient(135deg, #f43f5e, #ec4899); border-radius: 20px 20px 0 0; display: flex; justify-content: space-between; align-items: center; }
            .vh-title { margin: 0; font-size: 18px; font-weight: bold; color: white; }
            .vh-close { background: rgba(255,255,255,0.2); border: none; color: #fff; width: 32px; height: 32px; border-radius: 50%; font-size: 20px; cursor: pointer; }
            .vh-content { flex: 1; overflow-y: auto; padding: 16px; }
            .vh-content::-webkit-scrollbar { width: 4px; }
            .vh-content::-webkit-scrollbar-thumb { background: #f43f5e; border-radius: 4px; }
            .vh-btn-scan { width: 100%; padding: 14px; background: #3b82f6; color: #fff; border: none; border-radius: 12px; font-size: 16px; font-weight: bold; margin-bottom: 16px; cursor: pointer; }
            .vh-card { background: #1e293b; border-radius: 12px; padding: 12px; margin-bottom: 12px; border: 1px solid #334155; }
            .vh-card video { width: 100%; border-radius: 8px; max-height: 180px; background: #000; margin-bottom: 10px; }
            .vh-card-url { font-size: 11px; color: #94a3b8; word-break: break-all; margin-bottom: 10px; background: #0f172a; padding: 8px; border-radius: 6px; border: 1px dashed #334155; }
            .vh-badge-type { display: inline-block; padding: 3px 8px; border-radius: 6px; font-size: 10px; font-weight: bold; margin-bottom: 8px; }
            .badge-mp4 { background: #10b981; color: white; }
            .badge-m3u8 { background: #6366f1; color: white; }
            .vh-card-actions { display: grid; grid-template-columns: 1fr 1fr; gap: 8px; }
            .vh-btn { padding: 10px 5px; border: none; border-radius: 8px; font-size: 13px; font-weight: bold; cursor: pointer; display: flex; align-items: center; justify-content: center; gap: 5px; color: white; transition: 0.2s; }
            .vh-btn:active { transform: scale(0.95); }
            .vh-btn-dl { background: #10b981; }
            .vh-btn-open { background: #3b82f6; }
            .vh-btn-copy { background: #f59e0b; }
            .vh-empty { text-align: center; color: #64748b; padding: 30px 0; }
        `;
        document.head.appendChild(css);

        const container = document.createElement('div');
        container.id = 'vh-container';
        container.innerHTML = `
            <button id="vh-fab">🎬 <span id="vh-badge">0</span></button>
            <div id="vh-panel">
                <div class="vh-header">
                    <h3 class="vh-title">🔥 Video Hunter (M3U8)</h3>
                    <button class="vh-close" id="vh-close">✕</button>
                </div>
                <div class="vh-content">
                    <button class="vh-btn-scan" id="vh-scan">🔍 Quét Lại Video</button>
                    <div id="vh-list"></div>
                </div>
            </div>
        `;
        document.body.appendChild(container);

        document.getElementById('vh-fab').addEventListener('click', togglePanel);
        document.getElementById('vh-close').addEventListener('click', togglePanel);
        document.getElementById('vh-scan').addEventListener('click', () => {
            scanDOMVideos();
            renderList();
        });

        function togglePanel() {
            isPanelOpen = !isPanelOpen;
            document.getElementById('vh-panel').classList.toggle('open', isPanelOpen);
            if (isPanelOpen) {
                scanDOMVideos();
                renderList();
            }
        }
    }

    function updateBadge() {
        const badge = document.getElementById('vh-badge');
        if (badge) {
            const count = videoSources.size;
            badge.innerText = count;
            badge.style.display = count > 0 ? 'flex' : 'none';
        }
    }

    function renderList() {
        const listDiv = document.getElementById('vh-list');
        if (!listDiv) return;

        const urls = Array.from(videoSources);
        if (urls.length === 0) {
            listDiv.innerHTML = '<div class="vh-empty">❌ Không tìm thấy video.<br>Hãy chơi video trước rồi bấm Quét Lại!</div>';
            return;
        }

        listDiv.innerHTML = '';
        urls.forEach((url) => {
            const isM3U8 = url.includes('.m3u8');
            const fileName = getFileName(url);

            const card = document.createElement('div');
            card.className = 'vh-card';
            
            // Tạo preview
            let previewHtml = `<div id="preview-${url.hashCode()}" style="margin-bottom:10px;"></div>`;
            
            // Badge thể loại
            let badgeHtml = isM3U8 ? `<span class="vh-badge-type badge-m3u8">📡 STREAM (M3U8)</span>` : `<span class="vh-badge-type badge-mp4">📹 DIRECT (MP4/WEBM)</span>`;

            // Tên nút tải
            let downloadBtnText = isM3U8 ? "⬇️ Tải MP4" : "⬇️ Tải Xuống";

            card.innerHTML = `
                ${previewHtml}
                ${badgeHtml}
                <div class="vh-card-url">🔗 ${url.substring(0, 120)}...</div>
                <div class="vh-card-actions">
                    <button class="vh-btn vh-btn-dl" data-url="${url}" data-type="${isM3U8 ? 'm3u8' : 'direct'}" data-name="${fileName}">${downloadBtnText}</button>
                    <button class="vh-btn vh-btn-open" data-url="${url}">🌐 Open Link</button>
                    <button class="vh-btn vh-btn-copy" data-url="${url}">📋 Copy Link</button>
                </div>
            `;
            listDiv.appendChild(card);

            // Khởi tạo Preview (Dùng HLS.js cho M3U8)
            const previewContainer = card.querySelector(`#preview-${url.hashCode()}`);
            const video = document.createElement('video');
            video.controls = true;
            video.style.cssText = 'width: 100%; border-radius: 8px; max-height: 180px; background: #000;';
            previewContainer.appendChild(video);

            if (isM3U8 && Hls.isSupported()) {
                const hls = new Hls();
                hls.loadSource(url);
                hls.attachMedia(video);
            } else if (isM3U8 && video.canPlayType('application/vnd.apple.mpegurl')) {
                video.src = url; // Safari hỗ trợ sẵn
            } else {
                video.src = url;
            }

            // Gán sự kiện nút bấm
            card.querySelector('.vh-btn-dl').addEventListener('click', function() {
                if (this.dataset.type === 'm3u8') {
                    convertM3U8ToMP4(this.dataset.url, this);
                } else {
                    forceDownload(this.dataset.url, this.dataset.name, this);
                }
            });

            card.querySelector('.vh-btn-open').addEventListener('click', function() {
                window.open(this.dataset.url, '_blank');
            });

            card.querySelector('.vh-btn-copy').addEventListener('click', function() {
                copyToClipboard(this.dataset.url);
            });
        });
    }

    // Hàm băm String đơn giản để tạo ID
    String.prototype.hashCode = function() {
        let hash = 0;
        for (let i = 0; i < this.length; i++) {
            const char = this.charCodeAt(i);
            hash = ((hash << 5) - hash) + char;
            hash |= 0;
        }
        return Math.abs(hash);
    };

    // 6. KHỞI TẠO
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', createUI);
    } else {
        createUI();
    }

    document.addEventListener('play', function(e) {
        if (e.target.tagName === 'VIDEO') {
            scanDOMVideos();
            updateBadge();
            if (isPanelOpen) renderList();
        }
    }, true);

    setInterval(() => {
        scanDOMVideos();
    }, 3000);

})();
