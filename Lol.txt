// ==UserScript==
// @name         🌑 Video Hunter Ultimate - Smart M3U8
// @namespace    http://tampermonkey.net/

// @description  Tải M3U8->MP4 không lỗi, chỉ chạy khi có video, UI Monochrome

// @grant        GM_xmlhttpRequest
// @grant        GM_download
// @connect      *
// @run-at       document-idle

(function() {
    'use strict';

    let videoSources = new Set();
    let isPanelOpen = false;
    let hasVideoOnPage = false;
    let isInitialized = false;

    // ==================== SMART INIT (Chỉ chạy khi có video) ====================
    function checkForVideos() {
        // Kiểm tra có video element hoặc video URL không
        const hasVideoElement = document.querySelector('video') !== null;
        const hasVideoURL = document.body.innerHTML.match(/\.(mp4|webm|m3u8)/i) !== null;
        
        if (hasVideoElement || hasVideoURL) {
            hasVideoOnPage = true;
            if (!isInitialized) {
                console.log('🎬 Video Hunter: Phát hiện video, khởi động...');
                initializeScript();
            }
        }
    }

    // Chờ trang load xong mới kiểm tra
    if (document.readyState === 'loading') {
        document.addEventListener('DOMContentLoaded', checkForVideos);
    } else {
        checkForVideos();
    }

    // Kiểm tra lại sau 2s (cho trang AJAX)
    setTimeout(checkForVideos, 2000);

    function initializeScript() {
        if (isInitialized) return;
        isInitialized = true;
        
        setupNetworkInterceptors();
        createUI();
        startVideoMonitoring();
    }

    // ==================== NETWORK INTERCEPTORS ====================
    function setupNetworkInterceptors() {
        const originalFetch = window.fetch;
        window.fetch = function(...args) {
            const url = typeof args[0] === 'string' ? args[0] : (args[0]?.url || '');
            checkAndAddUrl(url);
            return originalFetch.apply(this, args);
        };

        const originalXHROpen = XMLHttpRequest.prototype.open;
        XMLHttpRequest.prototype.open = function(method, url, ...rest) {
            checkAndAddUrl(url);
            return originalXHROpen.apply(this, [method, url, ...rest]);
        };
    }

    function checkAndAddUrl(url) {
        if (!url || typeof url !== 'string' || url.startsWith('blob:')) return;
        if (/\.(mp4|webm|ogg|mov|m3u8|mpd|ts|flv)/i.test(url) || /(video|stream|videoplayback)/i.test(url)) {
            videoSources.add(url);
            updateBadge();
        }
    }

    // ==================== DOM SCANNER ====================
    function scanDOMVideos() {
        document.querySelectorAll('video').forEach(v => {
            if (v.src && !v.src.startsWith('blob:')) videoSources.add(v.src);
            if (v.currentSrc && !v.currentSrc.startsWith('blob:')) videoSources.add(v.currentSrc);
            v.querySelectorAll('source').forEach(s => { if (s.src) videoSources.add(s.src); });
        });
        
        // Quét data attributes
        document.querySelectorAll('[data-src], [data-video], [data-video-src], [data-url]').forEach(el => {
            const src = el.dataset.src || el.dataset.video || el.dataset.videoSrc || el.dataset.url;
            if (src && /\.(mp4|webm|m3u8)/i.test(src)) {
                videoSources.add(src);
            }
        });
    }

    function startVideoMonitoring() {
        // Lắng nghe sự kiện video play
        document.addEventListener('play', function(e) {
            if (e.target.tagName === 'VIDEO') {
                scanDOMVideos();
                updateBadge();
                if (isPanelOpen) renderList();
            }
        }, true);

        // Quét định kỳ (mỗi 3s)
        setInterval(scanDOMVideos, 3000);
    }

    // ==================== M3U8 TO MP4 CONVERTER (NÂNG CẤP) ====================
    async function convertM3U8ToMP4(m3u8Url, btnElement) {
        const originalText = btnElement.innerText;
        
        try {
            // Bước 1: Đọc playlist
            btnElement.innerText = "⏳ Đọc playlist...";
            btnElement.style.opacity = "0.6";
            
            const playlistText = await fetchWithFallback(m3u8Url);
            const tsUrls = parseM3U8Playlist(playlistText, m3u8Url);

            if (tsUrls.length === 0) {
                throw new Error("Playlist trống hoặc không hợp lệ");
            }

            console.log(`🎬 Tìm thấy ${tsUrls.length} đoạn video`);

            // Bước 2: Tải từng đoạn TS
            const chunks = [];
            for (let i = 0; i < tsUrls.length; i++) {
                btnElement.innerText = `⏳ Tải ${i+1}/${tsUrls.length}`;
                
                try {
                    const chunkData = await fetchWithFallback(tsUrls[i], true);
                    chunks.push(chunkData);
                } catch (err) {
                    console.warn(`⚠️ Lỗi tải đoạn ${i+1}, bỏ qua:`, err.message);
                    // Tiếp tục tải các đoạn còn lại
                }
            }

            if (chunks.length === 0) {
                throw new Error("Không tải được đoạn nào");
            }

            // Bước 3: Ghép file
            btnElement.innerText = "📦 Ghép video...";
            const finalBlob = new Blob(chunks, { type: 'video/mp2t' }); // TS format
            const blobUrl = URL.createObjectURL(finalBlob);

            // Bước 4: Tải xuống
            const a = document.createElement('a');
            a.href = blobUrl;
            a.download = `video_${Date.now()}.ts`; // Lưu dạng .ts (có thể đổi tên .mp4 sau)
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            
            setTimeout(() => URL.revokeObjectURL(blobUrl), 100);

            btnElement.innerText = "✅ Xong!";
            btnElement.style.background = "#ffffff";
            btnElement.style.color = "#000000";
            
            showToast(`✅ Đã tải ${chunks.length}/${tsUrls.length} đoạn! File lưu dạng .ts (đổi tên thành .mp4 để xem)`);

        } catch (error) {
            console.error("❌ Lỗi M3U8:", error);
            btnElement.innerText = "❌ Lỗi";
            btnElement.style.background = "#ff4d4d";
            btnElement.style.color = "#ffffff";
            
            showToast("❌ " + error.message + ". Hãy Copy Link rồi dùng IDM/1DM/yt-dlp!");
        }

        setTimeout(() => { 
            btnElement.innerText = originalText; 
            btnElement.style.opacity = "1"; 
            btnElement.style.background = ""; 
            btnElement.style.color = ""; 
        }, 4000);
    }

    // ==================== PARSE M3U8 PLAYLIST (Xử lý cả Master Playlist) ====================
    function parseM3U8Playlist(text, baseUrl) {
        const lines = text.split('\n').map(l => l.trim()).filter(l => l);
        const tsUrls = [];
        let base = baseUrl.substring(0, baseUrl.lastIndexOf('/') + 1);

        // Kiểm tra nếu là Master Playlist (chứa #EXT-X-STREAM-INF)
        if (text.includes('#EXT-X-STREAM-INF')) {
            console.log('📡 Phát hiện Master Playlist, tìm playlist con...');
            // Lấy playlist đầu tiên (thường là chất lượng cao nhất)
            for (let i = 0; i < lines.length; i++) {
                if (lines[i].startsWith('#EXT-X-STREAM-INF')) {
                    const nextLine = lines[i + 1];
                    if (nextLine && !nextLine.startsWith('#')) {
                        const subPlaylistUrl = nextLine.startsWith('http') ? nextLine : base + nextLine;
                        console.log('📥 Tải playlist con:', subPlaylistUrl);
                        // Tải đệ quy (nhưng trong trường hợp này ta ném lỗi để user biết)
                        throw new Error("Master Playlist phức tạp! Hãy bấm Open Link -> Xem source -> Tìm link .m3u8 cuối cùng");
                    }
                }
            }
        }

        // Parse Media Playlist (chứa .ts segments)
        for (let line of lines) {
            // Bỏ qua comment
            if (line.startsWith('#')) continue;
            
            // Chỉ lấy file .ts
            if (line.endsWith('.ts') || line.includes('.ts?')) {
                const url = line.startsWith('http') ? line : base + line;
                tsUrls.push(url);
            }
        }

        return tsUrls;
    }

    // ==================== FETCH WITH FALLBACK (Bypass CORS) ====================
    async function fetchWithFallback(url, isBlob = false) {
        // Thử fetch trực tiếp trước
        try {
            const response = await fetch(url, { 
                mode: 'cors', 
                credentials: 'omit',
                headers: { 'Origin': window.location.origin }
            });
            
            if (!response.ok) throw new Error(`HTTP ${response.status}`);
            
            return isBlob ? await response.blob() : await response.text();
        } catch (directError) {
            console.warn('⚠️ Fetch trực tiếp thất bại, thử CORS proxy...', directError.message);
            
            // Fallback: Dùng CORS proxy
            const proxyUrls = [
                `https://corsproxy.io/?${encodeURIComponent(url)}`,
                `https://api.allorigins.win/raw?url=${encodeURIComponent(url)}`
            ];

            for (let proxyUrl of proxyUrls) {
                try {
                    const response = await fetch(proxyUrl);
                    if (!response.ok) throw new Error(`Proxy ${response.status}`);
                    return isBlob ? await response.blob() : await response.text();
                } catch (proxyError) {
                    console.warn('⚠️ Proxy thất bại:', proxyError.message);
                }
            }

            // Nếu tất cả đều thất bại
            throw new Error("CORS blocked - Không thể tải");
        }
    }

    // ==================== FORCE DOWNLOAD MP4 ====================
    async function forceDownload(url, filename, btnElement) {
        const originalText = btnElement.innerText;
        btnElement.innerText = "⏳ Tải...";
        btnElement.style.opacity = "0.6";
        
        try {
            const blob = await fetchWithFallback(url, true);
            const blobUrl = URL.createObjectURL(blob);
            
            const a = document.createElement('a');
            a.href = blobUrl;
            a.download = filename;
            document.body.appendChild(a);
            a.click();
            document.body.removeChild(a);
            
            setTimeout(() => URL.revokeObjectURL(blobUrl), 100);

            btnElement.innerText = "✅ Xong!";
            btnElement.style.background = "#ffffff";
            btnElement.style.color = "#000000";
            showToast("✅ Đã tải xong!");
            
        } catch (error) {
            console.error("❌ Lỗi tải:", error);
            btnElement.innerText = "⚠️ Dùng Open";
            btnElement.style.background = "#ff4d4d";
            btnElement.style.color = "#ffffff";
            showToast("❌ Bị chặn CORS. Bấm 'Open' hoặc 'Copy Link'!");
        }
        
        setTimeout(() => { 
            btnElement.innerText = originalText; 
            btnElement.style.opacity = "1"; 
            btnElement.style.background = ""; 
            btnElement.style.color = ""; 
        }, 3000);
    }

    // ==================== UTILITIES ====================
    function copyToClipboard(text) {
        navigator.clipboard.writeText(text).then(() => showToast("✅ Đã copy!")).catch(() => {
            const textarea = document.createElement('textarea');
            textarea.value = text;
            document.body.appendChild(textarea);
            textarea.select();
            document.execCommand('copy');
            document.body.removeChild(textarea);
            showToast("✅ Đã copy!");
        });
    }

    function getFileName(url) {
        try {
            let path = new URL(url).pathname;
            let name = path.split('/').pop().split('?')[0];
            return name || 'video.mp4';
        } catch(e) { return 'video.mp4'; }
    }

    function showToast(message) {
        const toast = document.createElement('div');
        toast.innerText = message;
        toast.style.cssText = `position: fixed; bottom: 100px; left: 50%; transform: translateX(-50%); background: rgba(255,255,255,0.95); color: #000; padding: 12px 24px; border-radius: 30px; z-index: 2147483647; font-size: 13px; font-weight: 600; box-shadow: 0 4px 15px rgba(0,0,0,0.3); transition: opacity 0.5s; max-width: 80%;`;
        document.body.appendChild(toast);
        setTimeout(() => { toast.style.opacity = '0'; setTimeout(() => toast.remove(), 500); }, 3000);
    }

        if (badge) {

    function renderList() {
        const listDiv = document.getElementById('vh-list');
        if (!listDiv) return;

        const urls = Array.from(videoSources).filter(u => u.toLowerCase().includes(query));

            listDiv.innerHTML = `<div class="vh-empty">❌ Không tìm thấy video.<br>${videoSources.size > 0 ? 'Thử đổi từ khóa tìm kiếm.' : 'Phát video trên trang rồi quét lại.'}</div>`;

                <div class="vh-card-badges">${badgeHtml}</div>
                <div class="vh-card-url">${url}</div>
                <div class="vh-card-actions">

            `;
            listDiv.appendChild(card);
        });
